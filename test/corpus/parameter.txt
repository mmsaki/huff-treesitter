#define macro TRANSFER(to_addr) = takes (2) returns (1) {
    // to_addr should be an address type
    <to_addr>     // [to, amount]
    address       // [caller, to, amount]
    swap1         // [to, caller, amount]
    push0         // [0, to, caller, amount]
}
